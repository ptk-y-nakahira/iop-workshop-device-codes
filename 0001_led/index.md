```cpp
// ESP32でLEDを点滅させるプログラムです

// 準備するもの
// - ESP32
// - LED
// - 抵抗
// - ジャンパーワイヤー
// - ブレッドボード
// - USBケーブル

// 手順
// 1. ESP32の32番ピンとGNDピンを使用します
// 2. 32番ピンを、抵抗を挟んでLEDのアノード(LEDの長い脚)  に接続し、カソード（LEDの短い脚）をGNDに接続します
// 3. このプログラムを書き込むと、LEDが1秒間点灯し、1秒間消灯を繰り返します

#define LED_PIN 32 // ピン番号は配線と合っていれば何番でも大丈夫です。

void setup() {
  /* ここには一度だけ実行される初期化処理を書きます */

  pinMode(LED_PIN, OUTPUT); // LED_PIN=32を出力として設定します。
}

void loop() {
  /* ここに繰り返し処理を書きます */

  digitalWrite(LED_PIN, HIGH); // LED_PIN=32をHIGHにします。
                               // HIGHとは・・・
                               // 電圧がかかる状態です。
                               // LEDは電圧がかかると点灯します。

  delay(1000); // 1000ミリ秒＝1秒間点灯します。


  digitalWrite(LED_PIN, LOW); // LED_PIN=32をLOWにします。
                              // LOWとは・・・
                              // 電圧がかかっていない状態です
                              // LEDは電圧がかかっていないと消灯します


  // 1秒間消灯します。
  delay(1000);

  // これを繰り返すことで、1秒間点灯、1秒間消灯を繰り返します。
}
```
